14012. 合并果子

题面

https://acm.sjtu.edu.cn/OnlineJudge/problem?problem_id=14012

分析

注意⚠️！这里的题目要读清楚：合并的果子并不一定要是相邻的！这不是动态规划问题！

那一种简单的想法就是构建HuffmanTree，按照其构造的方式，可以实现让越小的数在计算过程中被加的次数最多，而越大的数则越小。

#include "iostream"
#include "limits.h"
using namespace std;

struct HuffmanNode
{
    long weight;
    long long parent;
    long long left, right;
};

long long minIndex(HuffmanNode Bt[], long long k, long long m)
{   long long i, min, minWeight = INT_MAX;//用一个不可能且很大的权值
    for (i=m-1; i>k; i--)
    {
        if ((Bt[i].parent==0)&&(Bt[i].weight < minWeight))
        {   min = i;
            minWeight = Bt[min].weight;
        }
    }
    return min;
}

long BestBinaryTree (long w[], long long n)
{
    long ans = 0;
    HuffmanNode *BBTree;
    long long first_min, second_min; //
    long long m=n*2; //共2n-1个结点，下标为0处不放结点
    long long i, j;
    //!在数组中从后往前存储叶子结点,倒放
    BBTree = new HuffmanNode[m];
    for (j=0; j<n; j++)
    {   i = m-1-j;
        BBTree[i].weight = w[j];
        BBTree[i].parent = 0;
        BBTree[i].left = 0;
        BBTree[i].right = 0;
    }
    i = n-1; // i is the position which is ready for the first new node
    while (i!=0) //数组左侧尚有未用空间，即新创建的结点个数还不足
    {   first_min = minIndex(BBTree, i, m);
        BBTree[first_min].parent = i;
        second_min = minIndex(BBTree, i, m);
        BBTree[second_min].parent = i;
        BBTree[i].weight = BBTree[first_min].weight + BBTree[second_min].weight;
        BBTree[i].parent = 0;
        BBTree[i].left = first_min; BBTree[i].right = second_min;
        ans += BBTree[i].weight;
        i--;  // !每次都要更新，新的权重加入到BBTree
    }
    return ans;
}


int main(){
    long n = 0;
    long w[10010];
    cin >> n;
    for(long i = 0; i < n; ++ i){
        cin >> w[i];
    }

    cout << BestBinaryTree(w, n);

}

时间复杂度是 $O(n^2)$

另一种方法是用小根堆来实现，先把果子全部入堆，然后每次连续pop两次，得到当前数目最小的两堆果子$a,b$，然后把 $a+b$ 放入堆中，对堆进行更新。

#include <iostream>
using namespace std;
const int N = 10010;
int huffman[N]; // 因为和huffman的过程很像，就直接用huffman命名吧
int num = 0;
// 插入到小根堆
void insert(int x){
    huffman[++num] = x;
    int now = num, nxt = num >> 1;
    while(nxt){
        if(huffman[now] >= huffman[nxt]) break;
        swap(huffman[now], huffman[nxt]);
        now = nxt;
        nxt >>= 1;
    }
}

int top(){
    return huffman[1];
}

// 删除小根堆
void erase(){
    huffman[1] = huffman[num--];
    int now = 1, nxt = 2;
    while(nxt <= num){
        if(nxt < num && huffman[nxt] > huffman[nxt+1]) ++nxt;
        if(huffman[now] <= huffman[nxt]) break;
        swap(huffman[now], huffman[nxt]);
        now = nxt;
        nxt <<= 1;
    }
}

int main(){
    int n;
    cin >> n;
    int tmp;
    for(int i = 0; i < n; ++i){
        cin >> tmp;
        insert(tmp);
    }

    int ans = 0;
    while(--n){
        int a = top(); erase();
        int b = top(); erase();
        ans += a + b;
        insert(a + b);
    }

    cout << ans;
}

插入和删除的时间复杂度均为$log(n)$

1049. 二哥学二叉树

题面

https://acm.sjtu.edu.cn/OnlineJudge/problem?problem_id=1049

分析

对于一个二叉树，假如我们知道它的前序遍历和中序遍历，它就能被唯一确定。我们是怎么确定的呢？

  1. 找到前序遍历第一个节点在中序节点中的位置 $i$
  2. $i$ 左边为该节点的左子树，右边为右子树
  3. 对两子树进行同样的操作

题目要求其实是用数组方式打印二叉树的层次遍历, 那么我们要利用完全二叉树中左右孩子和父亲i在层次遍历中满足的下标关系： 左孩子是 2i, 右孩子是 2i + 1。

全局变量的妙用
在 C++ 中，除非明确初始化数字全局变量，否则它们将自动初始化为零。全局字符变量初始化为 NULL。全局int,long, double等自动初始化为0。
#include <iostream>
#include <cstring>
using namespace std;
char pre[100],mid[100],ans[2000]; // 由于ans为全局变量，其初始化时会自动把所有位置设为空值！
void dfs(int pre_start, int pre_end, int mid_start, int mid_end, int index, int& max)
{
    int root, l, r; // 找前序遍历首位（root）在中序遍历中的下标
    for (root = mid_start;root < mid_end;++root){
        if (mid[root]==pre[pre_start])
            break;
    }
    ans[index] = mid[root];
    if (index > max) max = index; // 这是为了在后面打印数组中元素时知道何时停止
    l = root-mid_start;
    // 左右孩子和父亲i在层次遍历中满足的下标关系： 左孩子是 2i, 右孩子是 2i + 1
    if (l>0) dfs(pre_start+1,pre_start+l,mid_start,root,index * 2, max);
    r=mid_end-root-1;
    if (r>0) dfs(pre_start+l+1,pre_end,root+1,mid_end,index*2+1, max);
}
int main() {
    int max = 0; // 记载数组的大小
    cin >> pre;
    cin >> mid;
    dfs(0,strlen(pre),0,strlen(mid),1, max);
    for (int i=1;i <= max;++i)
        if (ans[i]) cout<<ans[i]<<" ";
        else cout<<"NULL ";
    return 0;
}

时间复杂度$O(n^2)$

ans[index] = mid[root]; 二叉树有n节点就会运行到n次, 上面的搜索root是$O(n)$

1223. Max Fraction

题面

https://acm.sjtu.edu.cn/OnlineJudge/problem?problem_id=1223

分析

解法一

我们看见这是求“第k大”的问题，于是想到大根堆。下面潦草地说一下算法过程。

算法的正确性可以这样证明（提供自同专业的朱同学）：

#include <iostream>
#include <queue>
using namespace std;

int n, k;

struct node {
    int numerator, denominator;
    bool operator < (const node &tmp) const {
        return 1.0 * numerator / denominator < 1.0 * tmp.numerator / tmp.denominator;
    }
};

int gcd(int numerator, int denominator) {
    return !denominator ? numerator : gcd(denominator, numerator % denominator);
}

int main(){
    cin >> n >> k;
    priority_queue<node> heap; // 大根堆 or 优先队列
    for (int i = 2; i <= n; ++i) {
        heap.push((node) {i - 1, i}); // 初始化堆内容
    }

    int numerator, denominator;
    while (k--) {
        node tmp = heap.top();
        heap.pop();
        numerator = tmp.numerator; denominator = tmp.denominator;
        if (k == 0) break;
        for (int i = numerator - 1; i >= 1; --i) {
            if (gcd(i, denominator) == 1) {
                heap.push((node) {i, denominator});
                break;
            }
        }
    }
    cout << numerator << '/' << denominator << endl;
    return 0;
}

时间复杂度 $klog(n)$

解法二

这道题其实有数学背景。

Farey Sequences（法雷数列）
对任意给定的一个自然数n，将分母小于等于n的不可约的真分数按升序排列，并且在第一个分数之前加上0/1，在最后一个分数之后加上1/1，这个序列称为n级法雷数列，以Fn表示。如F5为：0/1, 1/5, 1/4, 1/3, 2/5, 1/2, 3/5, 2/3, 3/4, 4/5, 1/1。

法雷数列的构造可采用二分法

即如果 a/b， c/d （a/b <c/d）是一个n级法雷数列中的两个元素，且b+d <=n, 则可以在a/b, c/d 中间插入一个分数 (a+b)/(c+d)。下面以5级法雷数列为例，给出详细的过程。

step1: 准备两个数 0/1, 1/1 作为整个法雷数列的第一个元素和最后一个元素
　　0/1, 1/1
step2: 在两个数中间插入1个数1/2, 变为
　　0/1, 1/2, 1/1
step3: 在每对相邻两个数中间插入1个数，变为
　　0/1, 1/3, 1/2, 2/3, 1/1
step4: 在每对相邻两个数中间插入1个数，变为
　　0/1, 1/4, 1/3, 2/5, 1/2, 3/5, 2/3, 3/4, 1/1
step5: 0/1 和 1/4 之间 和3/4和 1/1 仍然可插入1个数，使得插入的数分母不大于5
　　0/1, 1/5, 1/4, 1/3, 2/5, 1/2, 3/5, 2/3, 3/4, 4/5, 1/1

至此，该序列包含了所有分母不大于5的最简真分数，且各个分数以递增顺序排列。

法雷数列的构造还可采用递归关系

关于Farey序列的相邻3项，有如下性质：

若$\frac{p}{q}$在某法雷数列的邻项是$\frac{a}{b}$和$\frac{c}{d}$，即$\frac{a}{b}<\frac{p}{q}<\frac{c}{d}$

设$k=gcd(a+c,b+d)$，则可以计算出 $c=kp-a,d=kq-b$。

那么k应该取多少呢？注意到两点：

  • 分母必须 $\le n$；

  • k越大$\frac{c}{d}$越小（可用函数单调性证明），要准确的得到相邻三项中的第三项$\frac{c}{d}$， 必须使k取最大值。

由这两点可以得出 $k=max\{k | kq-b\le n\}=[(n+b)/q]$

#include<cstdio>
 
void farey(int n, int ans)
{
	int k, a0 = 0, b0 = 1, a1 = 1, b1 = n, a2, b2;
	for (int i = 0; i < ans - 1; ++i) /// 由于第1项已算出(1/n)，所以算ans-1次就行
	{
		k = (n + b0) / b1;///k要尽量取大
		a2 = k * a1 - a0;
		b2 = k * b1 - b0;///利用前两项计算出下一项
		a0 = a1, a1 = a2;
		b0 = b1, b1 = b2;///移动
	}
	printf("%d/%d\n", a1, b1);
}
 
int main()
{
	int n, ans;
	while (~scanf("%d%d", &n, &ans))
		farey(n, ans);
	return 0;
}

法雷数列有个性质：除了1级法雷数列外，所有的法雷数列都有奇数个元素，其中居于正中间的那个元素一定是1/2.

其实就是因为中间那个数是 1/2 ，而其他的数 $k$ 都可以在数组中找到 $1-k$，与之配对，所以总数总是奇数。

那在本题中如果要取第k大的数，只需要用 1 减去第k小的数就可以了。

#include<cstdio>

void farey(int n, int ans)
{
    int k, a0 = 0, b0 = 1, a1 = 1, b1 = n, a2, b2;
    for (int i = 0; i < ans - 1; ++i) /// 由于第1项已算出(1/n)，所以算ans-1次就行
    {
        k = (n + b0) / b1;///k要尽量取大
        a2 = k * a1 - a0;
        b2 = k * b1 - b0;///利用前两项计算出下一项
        a0 = a1, a1 = a2;
        b0 = b1, b1 = b2;///移动
    }
    printf("%d/%d\n", b1 - a1, b1);
}

int main()
{
    int n, ans;
    scanf("%d%d", &n, &ans);
    farey(n, ans);
    return 0;
}

时间复杂度$O(k)$ 这大概就是数学的魅力吧
