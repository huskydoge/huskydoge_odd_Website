看电影

题目描述

有一面废弃的墙面，虽然墙体上部凹凸不平，但是其侧面十分的光滑，可以用来投影看露天 电影。

现在，爱看电影的小 Z 想知道，在一个给定的墙面上，能够用来看电影的矩形区域的 最大面积。（小 Z 只在乎面积，他似乎并不在意这块墙的横纵比）

输入格式

第一行为一个整数 n，代表墙的长度。

第二行为用空格分隔的 n 个正整数，依次代表这段墙的高度。

输出格式

一个正整数，代表最大看电影面积。

输入输出样例

输入 #1复制

7
2 1 4 5 1 3 3

输出 #1复制

8

说明/提示

对30%的测试数据，n<=1000

对100%的测试数据，n<=1000000

每段墙的高度均为int范围内的正整数。

解答

单调栈

#include "iostream"
#include "stack"
using namespace std;

long n = 0;

long calArea(int* arr){
    stack<long> stack;
    long ans = 0;
    for (int i = 0; i <= n; i++) {
        int cur;
        if (i == n){
            cur = 0;
        }else {
            cur = arr[i];
        }

        while (!stack.empty() && cur <= arr[stack.top()]){
            int h = arr[stack.top()];
            stack.pop();
            long w = i;
            if (!stack.empty()){
                w = i - stack.top() - 1;
            }
            ans = max(ans ,h*w);
        }
        stack.push(i);
    }
    return ans;
}

int main(){

    cin >> n;
    if(n <= 0){
        cout << 0;
        return 0;
    }

    int a[n];
    for (long i=0; i < n; ++i){
        cin >> a[i];
    }

    cout << calArea(a);
}

分治法

#include <iostream>
#include <vector>
using namespace std;

void solve(vector<long long>& a, long long left, long long right, long long& ans) {
    if (left == right) {
        ans = max(ans, a[left]);
        return;
    }
    if (left > right) return;
    long long mid = (left + right) / 2;
    solve(a, left, mid, ans);
    solve(a, mid + 1, right, ans);
    long long h = a[mid];
    long long i = mid, j = mid;
    
    while (left < i && j < right) {
        if (a[i - 1] > a[j + 1]) {
            --i;
            h = min(h, a[i]);
            while (i > left && a[i - 1] >= h) --i;
            ans = max(ans, h * (j - i + 1));
        }
        else {
            ++j;
            h = min(h, a[j]);
            while (j < right && a[j + 1] >= h) ++j;
            ans = max(ans, h * (j - i + 1));
        }
    }
    if (j == right) {
        while (left <= i) {
            h = min(h, a[i]);
            ans = max(ans, h * (j - i + 1));
            --i;
        }
    }
    else {
        while (j <= right) {
            h = min(h, a[j]);
            ans = max(ans, h * (j - i + 1));
            ++j;
        }
    }
}

int main() {
    long long n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    long long ans = 0;
    solve(a, 0, n - 1, ans);
    cout << ans;
}

测核酸

题目描述

为了使问题更加简化，我们在一维的层面上讨论这个问题。假设需要覆盖的区域是一条长为n的直线，而在这些直线上有若干个核酸检测点。规定一个核酸检测点的覆盖半径是r，该点坐标为x，那么它能覆盖的区域就是[x-r,x+r]。注意，这里的区间都是指是离散的整数区间。为了方便统一化管理，所有的核酸检测点有相同的覆盖半径。

不过，由于不同的点位核酸需求量不同，有的点位可能会需要多个核酸测试点同时覆盖到。现在请问，能使得区间[1,n]所有的点都满足需求的最小覆盖半径是多少？

输入格式

第一行两个正整数n,m，分别代表直线长度以及核酸点位数目。

第二行有m个正整数，分别代表m个核酸测试点的坐标。

第三行有n个正整数，分别代表每个核算测试点需要被覆盖的数目。

输出格式

一个正整数，代表能实现全覆盖最小的覆盖半径。

输入输出样例

输入 #1

14 4
5 12 4 8
1 1 1 1 1 1 1 2 1 1 1 1 1 2

输出 #1

6

说明/提示

对于30%的测试数据，1<=n<=5000。

对于100%的测试数据，1<=n<=500000。

题目数据保证一定有解，并且保证1<=m<=n，不会有重复的核酸点位。

解答

二分法 + 滑动区间

#include <cstdio>
#include "algorithm"
using namespace std;

#define SIZE 500010

int n, m; // n:记录长度 m:核酸点位数
int isDetectPoint[SIZE]; // 记录该点是否为检查点
int needNum[SIZE]; // 记录每点的需要被覆盖的次数

bool check(int r) {
    int L = max(1 - r, 1); //避免越界
    int R = min(1 + r, n);
    int count = 0;

    for (int i = L; i <= R; i++) {
        count += isDetectPoint[i];
    }

    if (count < needNum[1]) return false;

    for (int i = 2; i <= n; i++) {
        int nl = max(1, i - r); // 更新区间
        int nr = min(n, i + r);
        while (L < nl) { // 新区间左边界在原先区间左边界的右边，一部分核酸点检测不到更新后的位置了
            count -= isDetectPoint[L];
            L++;
        }
        while (R < nr) { // 新区间左边界在原先区间左边界的右边，一部分新的核酸点可以检测
            R++;
            count += isDetectPoint[R];
        }
        if (count < needNum[i]) return false; // 如果点i在r下不能满足条件，就返回false
    }
    return true;
}


int main() {
    int pos = 0;
    scanf("%d %d", &n, &m);

    for (int i = 1; i <= m; i++) {
        scanf("%d", &pos);
        isDetectPoint[pos] = 1;
    }

    for (int i = 1; i <= n; i++) {
        scanf("%d", &needNum[i]);
    }

    int l = 0, r = n, mid = (l + r) >> 1;
    while (l <= r) {
        mid = (l + r) >> 1;
        if (check(mid)) {
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }

    printf("%d\n", l);
    return 0;
}
